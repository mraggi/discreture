119  static void construct_combination(combination& data, size_type m)
120  {
121  IntType k = data.size();

124  IntType r = k - i;

620  template <class P>
621  bool augment(combination& comb, P pred, IntType start = 0)
622  {
623  if (comb.empty())
624  {
625  if (start < m_n-m_k+1)
626  {
627  comb.push_back(start);
628  return true;
629  }
630  }
631  auto last = comb.back();
632  auto guysleft = m_k-comb.size();
633 
634  start = std::max(last+1,start);
635  for (size_t i = start; i < m_n-guysleft+1; ++i)
636  {
637  comb.push_back(i);
638  if (pred(comb))
639  return true;
640  comb.pop_back();
641  }
642  return false;
643  }

645  template <class P>
646  bool DFSUtil(combination& comb, P pred)
647  {
648  auto last = comb.back();
649 // auto currsize = comb.size();
650 
651  if (comb.size() < static_cast<size_t>(m_k))
652  {
653  if (augment(comb,pred))
654  return true;
655  }
656  // If it can't be augmented, be it because size is already k or else, we have to start backtracking
657  while (!comb.empty())
658  {
659  last = comb.back();
660  comb.pop_back();
661  if (augment(comb,pred,last+1))
662  return true;
663  }
664  return false;
